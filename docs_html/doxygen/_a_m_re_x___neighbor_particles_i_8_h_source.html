745  const long Cnt = 1;
746

748

751 
752  // Send.
753  for (int i = 0; i < num_rcvs; ++i) {
754  const int Who = neighbor_procs[i];
755  const long Cnt = 1;
756 
757  AMREX_ASSERT(Who >= 0 && Who < NProcs);
758 
759  ParallelDescriptor::Send(&snds[Who], Cnt, Who, SeqNum);
760  }

767 template <int NStructReal, int NStructInt>
768 void
769 NeighborParticleContainer<NStructReal, NStructInt>::
770 fillNeighborsMPI(bool reuse_rcv_counts) {
771 
772  BL_PROFILE("NeighborParticleContainer::fillNeighborsMPI");
773 
774 #ifdef BL_USE_MPI
775  const int NProcs = ParallelDescriptor::NProcs();
776 
777  // each proc figures out how many bytes it will send, and how
778  // many it will receive
779  if (!reuse_rcv_counts) getRcvCountsMPI();
780  if (num_snds == 0) return;
781 
782  Vector<int> RcvProc;
783  Vector<std::size_t> rOffset; // Offset (in bytes) in the receive buffer
784  std::size_t TotRcvBytes = 0;
785  for (int i = 0; i < NProcs; ++i) {
786  if (rcvs[i] > 0) {
787  RcvProc.push_back(i);
788  rOffset.push_back(TotRcvBytes);
789  TotRcvBytes += rcvs[i];
790  }
791  }
792 
793  const int nrcvs = RcvProc.size();
794  Vector<MPI_Status> stats(nrcvs);
795  Vector<MPI_Request> rreqs(nrcvs);
796 
797  const int SeqNum = ParallelDescriptor::SeqNum();
798 
799  // Allocate data for rcvs as one big chunk.
800  Vector<char> recvdata(TotRcvBytes);
801 
802  // Post receives.
803  for (int i = 0; i < nrcvs; ++i) {
804  const auto Who = RcvProc[i];
805  const auto offset = rOffset[i];
806  const auto Cnt = rcvs[Who];
807 
808  AMREX_ASSERT(Cnt > 0);
809  AMREX_ASSERT(Cnt < std::numeric_limits<int>::max());
810  AMREX_ASSERT(Who >= 0 && Who < NProcs);
811 
812  rreqs[i] = ParallelDescriptor::Arecv(&recvdata[offset], Cnt, Who, SeqNum).req();
813  }
814 
815  // Send.
816  for (const auto& kv : send_data) {
817  const auto Who = kv.first;
818  const auto Cnt = kv.second.size();
819 
820  AMREX_ASSERT(Cnt > 0);
821  AMREX_ASSERT(Who >= 0 && Who < NProcs);
822  AMREX_ASSERT(Cnt < std::numeric_limits<int>::max());
823 
824  ParallelDescriptor::Send(kv.second.data(), Cnt, Who, SeqNum);
825  }
826 
827  // unpack the received data and put them into the proper neighbor buffers
828  if (nrcvs > 0) {
829  ParallelDescriptor::Waitall(rreqs, stats);
830  for (int i = 0; i < nrcvs; ++i) {
831  const int offset = rOffset[i];
832  char* buffer = &recvdata[offset];
833  int num_tiles, lev, gid, tid, size, np;
834  std::memcpy(&num_tiles, buffer, sizeof(int)); buffer += sizeof(int);
835  for (int j = 0; j < num_tiles; ++j) {
836  std::memcpy(&lev, buffer, sizeof(int)); buffer += sizeof(int);
837  std::memcpy(&gid, buffer, sizeof(int)); buffer += sizeof(int);
838  std::memcpy(&tid, buffer, sizeof(int)); buffer += sizeof(int);
839  std::memcpy(&size, buffer, sizeof(int)); buffer += sizeof(int);
840 
841  if (size == 0) continue;
842 
843  np = size / cdata_size;
844 
845  BL_ASSERT(size % cdata_size == 0);
846 
847  PairIndex dst_index(gid, tid);
848  size_t old_size = neighbors[lev][dst_index].size();
849  size_t new_size = neighbors[lev][dst_index].size() + np;
850  neighbors[lev][dst_index].resize(new_size);
851 
852  char* dst = (char*) &neighbors[lev][dst_index][old_size];
853  char* src = buffer;
854 
855  for (int n = 0; n < np; ++n) {
856  for (int ii = 0; ii < AMREX_SPACEDIM + NStructReal; ++ii) {
857  if (rc[ii]) {
858  std::memcpy(dst, src, sizeof(typename ParticleType::RealType));
859  src += sizeof(typename ParticleType::RealType);
860  }
861  dst += sizeof(typename ParticleType::RealType);
862  }
863  for (int ii = 0; ii < 2 + NStructInt; ++ii) {
864  if (ic[ii]) {
865  std::memcpy(dst, src, sizeof(int));
866  src += sizeof(int);
867  }
868  dst += sizeof(int);
869  }
870  }
871  buffer += size;
872  }
873  }
874  }
875 #endif
876 }

878 template <int NStructReal, int NStructInt>
879 template <class CheckPair>
880 void
881 NeighborParticleContainer<NStructReal, NStructInt>::
882 buildNeighborList(CheckPair check_pair, bool sort) {
883 
884  BL_PROFILE("NeighborParticleContainer::buildNeighborList");
885  AMREX_ASSERT(this->OK());
886 
887  for (int lev = 0; lev < this->numLevels(); ++lev) {
888 
889  neighbor_list[lev].clear();
890 
891  for (MyParIter pti(*this, lev); pti.isValid(); ++pti) {
892  PairIndex index(pti.index(), pti.LocalTileIndex());
893  neighbor_list[lev][index];
894  }
895 
896  IntVect ref_fac = computeRefFac(0, lev);
897 
898 #ifdef _OPENMP
899 #pragma omp parallel
900 #endif
901  {
902 
903  Vector<IntVect> cells;
904  Vector<ParticleType> tmp_particles;
905  BaseFab<int> head;
906  Vector<int> list;
907 
908  for (MyParIter pti(*this, lev, MFItInfo().SetDynamic(true)); pti.isValid(); ++pti) {
909 
910  PairIndex index(pti.index(), pti.LocalTileIndex());
911 #ifdef AMREX_USE_CUDA
912  Cuda::HostVector<int> nl;
913 #else
914  IntVector& nl = neighbor_list[lev][index];
915 #endif
916  AoS& particles = pti.GetArrayOfStructs();
917 
918  int Np = particles.size();
919  int Nn = neighbors[lev][index].size();
920  int N = Np + Nn;
921 
922  cells.resize(N);
923  tmp_particles.resize(N);
924  std::memcpy(&tmp_particles[0], particles.data(), Np*sizeof(ParticleType));
925  if (Nn > 0)
926  std::memcpy(&tmp_particles[Np], neighbors[lev][index].dataPtr(), Nn*pdata_size);
927 
928  // For each cell on this tile, we build linked lists storing the
929  // indices of the particles belonging to it.
930  Box box = pti.tilebox();
931  box.coarsen(ref_fac);
932  box.grow(num_neighbor_cells+1); // need an extra cell to account for roundoff errors.
933  head.resize(box);
934  head.setVal(-1);
935  list.resize(N, -1);
936 
937  for (int i = 0; i < N; ++i) {
938  const ParticleType& p = tmp_particles[i];
939  const IntVect& cell = this->Index(p, 0); // we always bin on level 0
940  cells[i] = cell;
941  list[i] = head(cell);
942  head(cell) = i;
943  }
944 
945  // using these lists, we build a neighbor list containing both
946  // kinds of particles.
947  int p_start_index = 0;
948  for (unsigned i = 0; i < Np; ++i) {
949  const ParticleType& p = tmp_particles[i];
950 
951  int num_neighbors = 0;
952  nl.push_back(0);
953 
954  const IntVect& cell = cells[i];
955  Box bx(cell, cell);
956  bx.grow(num_neighbor_cells);
957 
958  for (IntVect iv = bx.smallEnd(); iv <= bx.bigEnd(); bx.next(iv)) {
959  int j = head(iv);
960  while (j >= 0) {
961  if (i == j) {
962  j = list[j];
963  continue;
964  }
965  if ( check_pair(p, tmp_particles[j]) ) {
966  nl.push_back(j+1);
967  num_neighbors += 1;
968  }
969  j = list[j];
970  }
971  }
972 
973  nl[p_start_index] = num_neighbors;
974  p_start_index += num_neighbors + 1;
975  }
976 
977  if (sort) {
978  for (unsigned i = 0; i < nl.size(); i += nl[i] +1) {
979 #ifdef AMREX_USE_CUDA
980  thrust::sort(nl.begin() + i + 1,
981  nl.begin() + nl[i] + i + 1);
982 #else
983  std::sort(nl.begin() + i + 1,
984  nl.begin() + nl[i] + i + 1);
985 #endif
986  }
987  }
988 #ifdef AMREX_USE_CUDA
989  neighbor_list[lev][index].resize(nl.size());
990  thrust::copy(nl.begin(), nl.end(), neighbor_list[lev][index].begin());
991 #endif
992  }
993  }
994  }
995 }
