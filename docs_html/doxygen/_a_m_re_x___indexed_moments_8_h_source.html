469 checkMoments(IndexedMoments<Dim, ORDER> & a_moments,
470  Vector<IndexTM<int, Dim> > & a_bogusPowers,
471  const Real & a_dx,
472  const Real & a_tolerance,

474  const bool & a_bindMoments)
475 {
476  MomentIterator<Dim, ORDER> momit;
477  a_bogusPowers.resize(0);
478  Real cellvol = 1;
479  for(int idir = 0; idir < Dim; idir++)
480  {
481  cellvol *= a_dx;
482  }
483 
484  //shift moments to lowest corner
485  IndexTM<Real, Dim> shiftvec;
486  shiftvec.setAll(0.5*a_dx);
487  a_moments.shift(shiftvec);
488 
489  //because they are all shifted, to the lowest
490  //min is always zero.
491  const Real zerval = 0;
492  for(momit.reset(); momit.ok(); ++momit)
493  {
494  const IndexTM<int, Dim>& p = momit();
495  //this makes max = dx^p
496  Real maxval = POW<Dim>(a_dx, p);
497  //because this is a vol integral...
498  maxval *= cellvol;
499  //this is a reference so it will change the input
500  Real& momval = a_moments[p];
501  if(momval < zerval)
502  {
503  if((zerval - momval) > a_tolerance*maxval)
504  {
505  a_bogusPowers.push_back(p);
506  // MayDay::Error("minval break low");
507  }
508  if(a_bindMoments)
509  {
510  momval = zerval;
511  }
512  }
513  else if((momval > maxval) && (!a_ebMoment))
514  {
515  if((momval - maxval) > a_tolerance*maxval)
516  {
517  a_bogusPowers.push_back(p);
518  // MayDay::Error("maxval break");
519  }
520  if(a_bindMoments)
521  {
522  momval = maxval;
523  }
524  }
525  }
526 
527  //shift moments back to highest corner
528  shiftvec.setAll(-a_dx);
529  a_moments.shift(shiftvec);
530 
531  //because the moments are shifted to the highest corner
532  //even powers will always be positive, odd ones will always be negtive
533  for(momit.reset(); momit.ok(); ++momit)
534  {
535  const IndexTM<int, Dim>& p = momit();
536  int psum = p.sum();
537  bool pEven = (psum % 2 == 0);
538  Real& momval = a_moments[p];
539  //this makes max = dx^p
540  Real maxval = POW<Dim>(a_dx, p);
541  //because this is a vol integral...
542  maxval *= cellvol;
543  if(pEven && (momval < zerval) && (!a_ebMoment))
544  {
545  //p has an sum and therefore the moment should be positive
546  if(momval < -a_tolerance*maxval )
547  {
548  a_bogusPowers.push_back(p);
549  // MayDay::Error("high corner break even");
550  }
551  if(a_bindMoments)

553  momval = 0;
554  }
555  }

557  {
