907 template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
908 void
909 ParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>
910 ::CreateVirtualParticles (int level, AoS& virts) const
911 {
912  BL_PROFILE("ParticleContainer<NSR, NSI, NAR, NAI>::CreateVirtualParticles()");
913  BL_ASSERT(level > 0);
914  BL_ASSERT(virts.empty());
915 
916  if (level >= static_cast<int>(m_particles.size()))
917  return;
918 
919  if (aggregation_type == "")
920  {
921  ParmParse pp("particles");
922  aggregation_type = "None";
923  pp.query("aggregation_type", aggregation_type);
924  aggregation_buffer = 2;
925  pp.query("aggregation_buffer", aggregation_buffer);
926  }
927 
928  if (aggregation_type == "None");
929  else if (aggregation_type == "Cell");
930  else if (aggregation_type == "Flow") amrex::Abort("Flow aggregation not implemented");
931  else amrex::Abort("Unknown Particle Aggregation mode");
932 
933  if (aggregation_type == "None")
934  {
935  const auto& pmap = m_particles[level];
936  for (const auto& kv : pmap)
937  {
938  const auto& pbox = kv.second.GetArrayOfStructs();
939  for (auto it = pbox.cbegin(); it != pbox.cend(); ++it)
940  {
941  virts.push_back(*it);
942  virts.back().m_idata.id = VirtualParticleID;
943  }
944  }
945  return;
946  }
947 
948 
949  if (aggregation_type == "Cell")
950  {
951  BoxList bl_buffer;
952  bl_buffer.complementIn(Geom(level).Domain(), ParticleBoxArray(level));
953  BoxArray buffer(std::move(bl_buffer));
954  buffer.grow(aggregation_buffer);
955 
956  const auto& pmap = m_particles[level];
957  for (const auto& kv : pmap)
958  {
959  const auto& pbox = kv.second.GetArrayOfStructs();
960 
961  std::map<IntVect,ParticleType> agg_map;
962 
963  for (auto it = pbox.cbegin(); it != pbox.cend(); ++it)
964  {
965  IntVect cell = Index(*it, level);
966  if (buffer.contains(cell))
967  {
968  // It's in the no-aggregation buffer.
969  // Set its id to indicate that it's a virt.
970  virts.push_back(*it);
971  virts.back().m_idata.id = VirtualParticleID;
972  }
973  else
974  {
975  //
976  // Note that Cell aggregation assumes that p.m_rdata.arr[AMREX_SPACEDIM] is mass and
977  // that all other components should be combined in a mass-weighted
978  // average.
979  //
980  auto agg_map_it = agg_map.find(cell);
981 
982  if (agg_map_it == agg_map.end())
983  {
984  //
985  // Add the particle.
986  //
987  ParticleType p = *it;
988  //
989  // Set its id to indicate that it's a virt.
990  //
991  p.m_idata.id = VirtualParticleID;
992  agg_map[cell] = p;
993  }
994  else
995  {
996  BL_ASSERT(agg_map_it != agg_map.end());
997  const ParticleType& pnew = *it;
998  ParticleType& pold = agg_map_it->second;
999  const Real old_mass = pold.m_rdata.arr[AMREX_SPACEDIM];
1000  const Real new_mass = pnew.m_rdata.arr[AMREX_SPACEDIM];
1001  const Real total_mass = old_mass + new_mass;
1002  //
1003  // Set the position to the center of mass.
1004  //
1005  for (int i = 0; i < AMREX_SPACEDIM; i++)
1006  {
1007  pold.m_rdata.pos[i] = (old_mass*pold.m_rdata.pos[i] + new_mass*pnew.m_rdata.pos[i])/total_mass;
1008  }
1009  BL_ASSERT(this->Index(pold, level) == cell);
1010  //
1011  // Set the metadata (presumably velocity) to the mass-weighted average.
1012  //
1013  for (int i = AMREX_SPACEDIM + 1; i < AMREX_SPACEDIM + NStructReal; i++)
1014  {
1015  pold.m_rdata.arr[i] = (old_mass*pold.m_rdata.arr[i] + new_mass*pnew.m_rdata.arr[i])/total_mass;
1016  }
1017  pold.m_rdata.arr[AMREX_SPACEDIM] = total_mass;
1018  }
1019  }
1020  }
1021 
1022  //
1023  // Add the aggregated particles to the virtuals.
1024  //
1025  for (const auto& agg_particle : agg_map)
1026  {
1027  virts.push_back(agg_particle.second);
1028  }
1029  }
1030  }
1031 }
