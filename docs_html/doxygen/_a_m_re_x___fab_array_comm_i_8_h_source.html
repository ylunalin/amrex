630  dfab->copyFromMem(bx, scomp, ncomp, dptr);
631  dptr += bx.numPts() * ncomp * sizeof(value_type);
632  }
633  BL_ASSERT(dptr == fb_recv_data[k] + fb_recv_size[k]);
634  }
635  }
636  }
637 
638  if (fb_the_recv_data)
639  {
640  if (ParallelDescriptor::MPIOneSided()) {
641 #if defined(BL_USE_MPI3)
642  MPI_Win_detach(ParallelDescriptor::fb_win, fb_the_recv_data);
643 #endif
644  }
645  amrex::The_FA_Arena()->free(fb_the_recv_data);
646  fb_the_recv_data = nullptr;
647  }
648  }
649 
650  if (N_snds > 0) {
651  if (!ParallelDescriptor::MPIOneSided()) {
652  Vector<MPI_Status> stats;
653  FabArrayBase::WaitForAsyncSends(N_snds,fb_send_reqs,fb_send_data,stats);
654  }
655  amrex::The_FA_Arena()->free(fb_the_send_data);
656  fb_the_send_data = nullptr;
657  }
658 
659 #ifdef BL_USE_TEAM
660  ParallelDescriptor::MyTeam().MemoryBarrier();
661 #endif
662 
663 #endif // MPI
664 }

666 template <class FAB>
667 void
668 FabArray<FAB>::ParallelCopy (const FabArray<FAB>& src,
669  int scomp,
670  int dcomp,
671  int ncomp,
672  const IntVect& snghost,
673  const IntVect& dnghost,
674  const Periodicity& period,
675  CpOp op,
676  const FabArrayBase::CPC * a_cpc)
677 {
678  BL_PROFILE("FabArray::ParallelCopy()");
679 
680  if (size() == 0 || src.size() == 0) return;
681 
682  BL_ASSERT(op == FabArrayBase::COPY || op == FabArrayBase::ADD);
683  BL_ASSERT(boxArray().ixType() == src.boxArray().ixType());
684 
685  BL_ASSERT(src.nGrowVect().allGE(snghost));
686  BL_ASSERT( nGrowVect().allGE(dnghost));
687 
688  if ((src.boxArray().ixType().cellCentered() || op == FabArrayBase::COPY) &&
689  (boxarray == src.boxarray && distributionMap == src.distributionMap)
690  && snghost == IntVect::TheZeroVector() && dnghost == IntVect::TheZeroVector()
691  && !period.isAnyPeriodic())
692  {
693  //
694  // Short-circuit full intersection code if we're doing copy()s or if
695  // we're doing plus()s on cell-centered data. Don't do plus()s on
696  // non-cell-centered data this simplistic way.
697  //
698 #ifdef _OPENMP
699 #pragma omp parallel if (FAB::isCopyOMPSafe() && Gpu::notInLaunchRegion())
700 #endif
701  for (MFIter fai(*this,TilingIfNotGPU()); fai.isValid(); ++fai)
702  {
703  const Box& bx = fai.tilebox();
704 
705  // avoid self copy or plus
706  if (this != &src) {
707  auto const sfab = src.array(fai);
708  auto dfab = this->array(fai);
709  if (op == FabArrayBase::COPY) {
710  AMREX_HOST_DEVICE_FOR_4D ( bx, ncomp, i, j, k, n,
711  {
712  dfab(i,j,k,dcomp+n) = sfab(i,j,k,scomp+n);
713  });
714  } else {
715  AMREX_HOST_DEVICE_FOR_4D ( bx, ncomp, i, j, k, n,
716  {
717  dfab(i,j,k,dcomp+n) += sfab(i,j,k,scomp+n);
718  });
719  }
720  }
721  }
722 
723  return;
724  }
725 
726  const CPC& thecpc = (a_cpc) ? *a_cpc : getCPC(dnghost, src, snghost, period);
727 
728  if (ParallelContext::NProcsSub() == 1)
729  {
730  //
731  // There can only be local work to do.
732  //
733  int N_loc = (*thecpc.m_LocTags).size();
734  bool is_thread_safe = FAB::isCopyOMPSafe() && thecpc.m_threadsafe_loc;
735  if (Gpu::inLaunchRegion() || !is_thread_safe)
736  {
737  LayoutData<Vector<FabCopyTag<FAB> > > loc_copy_tags(boxArray(),DistributionMap());
738  for (int i = 0; i < N_loc; ++i)
739  {
740  const CopyComTag& tag = (*thecpc.m_LocTags)[i];
741  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {
742  loc_copy_tags[tag.dstIndex].push_back
743  ({src.fabHostPtr(tag.srcIndex), tag.dbox, tag.sbox.smallEnd()-tag.dbox.smallEnd()});
744  }
745  }
746 #ifdef _OPENMP
747 #pragma omp parallel if (Gpu::notInLaunchRegion())
748 #endif
749  for (MFIter mfi(*this); mfi.isValid(); ++mfi)
750  {
751  const auto& tags = loc_copy_tags[mfi];
752  auto dfab = this->array(mfi);
753  if (op == FabArrayBase::COPY)
754  {
755  for (auto const & tag : tags)
756  {
757  auto const sfab = tag.sfab->array();
758  Dim3 offset = tag.offset.dim3();
759  AMREX_HOST_DEVICE_FOR_4D ( tag.dbox, ncomp, i, j, k, n,
760  {
761  dfab(i,j,k,dcomp+n) = sfab(i+offset.x,j+offset.y,k+offset.z,scomp+n);
762  });
763  }
764  }
765  else
766  {
767  for (auto const & tag : tags)
768  {
769  auto const sfab = tag.sfab->array();
770  Dim3 offset = tag.offset.dim3();
771  AMREX_HOST_DEVICE_FOR_4D ( tag.dbox, ncomp, i, j, k, n,
772  {
773  dfab(i,j,k,dcomp+n) += sfab(i+offset.x,j+offset.y,k+offset.z,scomp+n);
774  });
775  }
776  }
777  }
778  }
779  else
780  {
781 #ifdef _OPENMP
782 #pragma omp parallel for
783 #endif
784  for (int i = 0; i < N_loc; ++i)
785  {
786  const CopyComTag& tag = (*thecpc.m_LocTags)[i];
787  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {
788  // avoid self copy or plus
789  const FAB* sfab = &(src[tag.srcIndex]);
790  FAB* dfab = &(get(tag.dstIndex));
791  if (op == FabArrayBase::COPY)
792  {
793  dfab->copy(*sfab, tag.sbox, scomp, tag.dbox, dcomp, ncomp);
794  }
795  else
796  {
797  dfab->plus(*sfab, tag.sbox, tag.dbox, scomp, dcomp, ncomp);
798  }
799  }
800  }
801  }
802 
803  return;
804  }
805 
806 #ifdef BL_USE_MPI
807 
808 #if defined(BL_USE_MPI3)
809  BL_ASSERT(FAB::preAllocatable());
810 #else
811  BL_ASSERT(!ParallelDescriptor::MPIOneSided());
812 #endif
813 
814  //
815  // Do this before prematurely exiting if running in parallel.
816  // Otherwise sequence numbers will not match across MPI processes.
817  //
818  int preSeqNum;
819  if (!FAB::preAllocatable()) {
820  preSeqNum = ParallelDescriptor::SeqNum();
821  }
822  int SeqNum = ParallelDescriptor::SeqNum();
823 
824  const int N_snds = thecpc.m_SndTags->size();
825  const int N_rcvs = thecpc.m_RcvTags->size();
826  const int N_locs = thecpc.m_LocTags->size();
827 
828  if (N_locs == 0 && N_rcvs == 0 && N_snds == 0)
829  //
830  // No work to do.
831  //
832  return;
833 
834 #ifdef BL_USE_MPI3
835  MPI_Group tgroup, rgroup, sgroup;
836  if (ParallelDescriptor::MPIOneSided()) {
837  MPI_Comm_group(ParallelDescriptor::Communicator(), &tgroup);
838  }
839 #endif
840 
841  //
842  // Send/Recv at most MaxComp components at a time to cut down memory usage.
843  //
844  int NCompLeft = ncomp;
845 
846  for (int ipass = 0, SC = scomp, DC = dcomp; ipass < ncomp; )
847  {
848  const int NC = std::min(NCompLeft,FabArrayBase::MaxComp);
849 
850  //
851  // Before we post recv, let's preprocess sends in case FAB is not preAllocatable
852  //
853  char* the_send_data;
854  Vector<char*> send_data;
855  Vector<int> send_size;
856  Vector<int> send_rank;
857  Vector<MPI_Request> send_reqs;
858  Vector<const CopyComTagsContainer*> send_cctc;
859  Vector<Vector<int> > indv_send_size;
860  Vector<MPI_Request> pre_reqs;
861 
862 #if defined(BL_USE_MPI3)
863  int actual_n_snds = 0;
864 #endif
865  if (N_snds > 0)
866  {
867  send_data.reserve(N_snds);
868  send_size.reserve(N_snds);
869  send_rank.reserve(N_snds);
870  send_reqs.reserve(N_snds);
871  send_cctc.reserve(N_snds);
872  indv_send_size.reserve(N_snds);
873 
874  std::size_t total_volume = 0;
875  for (auto const& kv : *thecpc.m_SndTags)
876  {
877  Vector<int> iss;
878  auto const& cctc = kv.second;
879 
880  std::size_t nbytes = 0;
881  if (FAB::preAllocatable())
882  {
883  for (auto const& cct : kv.second)
884  {
885  nbytes += src[cct.srcIndex].nBytes(cct.sbox,SC,NC);
886  }
887  }
888  else
889  {
890  for (auto const& tag : cctc)
891  {
892  std::size_t b = src[tag.srcIndex].nBytes(tag.sbox,SC,NC);
893  nbytes += b;
894  iss.push_back(static_cast<int>(b));
895  }
896  }
897 
898  BL_ASSERT(nbytes < std::numeric_limits<int>::max());
899 
900  total_volume += nbytes;
901 
902  send_data.push_back(nullptr);
903  send_size.push_back(static_cast<int>(nbytes));
904  send_rank.push_back(kv.first);
905  send_reqs.push_back(MPI_REQUEST_NULL);
906  send_cctc.push_back(&cctc);
907  indv_send_size.push_back(std::move(iss));
908  }
909 
910  if (total_volume > 0)
911  {
912  the_send_data = static_cast<char*>(amrex::The_FA_Arena()->alloc(total_volume));
913  char* p = the_send_data;
914  for (int i = 0, N = send_size.size(); i < N; ++i) {
915  if (send_size[i] > 0) {
916  send_data[i] = p;
917  p += send_size[i];
918  }
919  }
920  }
921 
922 #if defined(BL_USE_MPI3)
923  actual_n_snds = N_snds - std::count(send_size.begin(), send_size.end(), 0);
924 #endif
925  }
926 
927  if (!FAB::preAllocatable())
928  {
929  pre_reqs.resize(N_snds,MPI_REQUEST_NULL);
930  for (int j=0; j<N_snds; ++j)
931  {
932  pre_reqs[j] = ParallelDescriptor::Asend
933  (indv_send_size[j].data(), indv_send_size[j].size(),
934  ParallelContext::global_to_local_rank(send_rank[j]),
935  preSeqNum,
936  ParallelContext::CommunicatorSub()).req();
937  }
938  }
939 
940  Vector<int> recv_from;
941  Vector<char*> recv_data;
942  Vector<int> recv_size;
943  Vector<MPI_Request> recv_reqs;
944 #ifdef BL_USE_MPI3
945  Vector<MPI_Aint> recv_disp;
946 #endif
947  //
948  // Post rcvs. Allocate one chunk of space to hold'm all.
949  //
950  char* the_recv_data = nullptr;
951 
952  int actual_n_rcvs = 0;
953  if (N_rcvs > 0) {
954  if (ParallelDescriptor::MPIOneSided()) {
955 #if defined(BL_USE_MPI3)
956  PostRcvs_MPI_Onesided(*thecpc.m_RcvTags, the_recv_data, recv_data,
957  recv_size, recv_from, recv_reqs, recv_disp,
958  SC, NC, SeqNum, ParallelDescriptor::cp_win);
959  MPI_Group_incl(tgroup, recv_from.size(), recv_from.dataPtr(), &rgroup);
960  MPI_Win_post(rgroup, 0, ParallelDescriptor::cp_win);
961 #endif
962  } else {
963  PostRcvs(*thecpc.m_RcvTags, the_recv_data,
964  recv_data, recv_size, recv_from, recv_reqs, SC, NC, SeqNum, preSeqNum);
965  }
966  actual_n_rcvs = N_rcvs - std::count(recv_size.begin(), recv_size.end(), 0);
967  }
968 
969  //
970  // Post send's
971  //
972  if (N_snds > 0)
973  {
974  bool is_thread_safe = FAB::isCopyOMPSafe();
975 #ifdef _OPENMP
976 #pragma omp parallel if (is_thread_safe && Gpu::notInLaunchRegion())
977 #endif
978  for (Gpu::StreamIter sit(N_snds,is_thread_safe); sit.isValid(); ++sit)
979  {
980  const int j = sit();
981  char* dptr = send_data[j];
982  if (dptr != nullptr)
983  {
984  auto const& cctc = *send_cctc[j];
985  for (auto const& tag : cctc)
986  {
987  const Box& bx = tag.sbox;
988  auto const sfab = src.array(tag.srcIndex);
989  auto pfab = amrex::makeArray4((value_type*)(dptr),bx);
990  AMREX_HOST_DEVICE_FOR_4D ( bx, NC, i, j, k, n,
991  {
992  pfab(i,j,k,n) = sfab(i,j,k,SC+n);
993  });
994 
995  dptr += (bx.numPts() * NC * sizeof(value_type));
996  }
997  BL_ASSERT(dptr == send_data[j] + send_size[j]);
998  }
999  }
1000 
1001  if (ParallelDescriptor::MPIOneSided())
1002  {
1003 #if defined(BL_USE_MPI3)
1004  Vector<MPI_Aint> send_disp(N_snds,0);
1005  send_reqs.clear();
1006 
1007  for (int i=0; i<N_snds; ++i) {
1008  if (send_size[i] > 0) {
1009  send_reqs[i] = ParallelDescriptor::Arecv
1010  (&send_disp[i],1,send_rank[i],SeqNum).req();
1011  }
1012  }
1013 
1014  MPI_Group_incl(tgroup, N_snds, send_rank.data(), &sgroup);
1015  MPI_Win_start(sgroup,0,ParallelDescriptor::cp_win);
1016 
1017  int send_counter = 0;
1018  while (send_counter < actual_n_snds) {
1019  MPI_Status status;
1020  int index;
1021 
1022  ParallelDescriptor::Waitany(send_reqs, index, status);
1023 
1024  BL_ASSERT(status.MPI_TAG == SeqNum);
1025  BL_ASSERT(status.MPI_SOURCE == send_rank[index]);
1026 
1027  MPI_Put(send_data[index], send_size[index], MPI_CHAR, send_rank[index],
1028  send_disp[index], send_size[index], MPI_CHAR, ParallelDescriptor::cp_win);
1029 
1030  ++send_counter;
1031  }
1032 #endif
1033  }
1034  else
1035  {
1036  int send_counter = 0;
1037  while (send_counter < N_snds)
1038  {
1039  int j;
1040 
1041  if (FAB::preAllocatable())
1042  {
1043  j = send_counter;
1044  }
1045  else
1046  {
1047  MPI_Status status;
1048  ParallelDescriptor::Waitany(pre_reqs, j, status);
1049  }
1050 
1051  if (send_size[j] > 0)
1052  {
1053  if (FabArrayBase::do_async_sends)
1054  {
1055  send_reqs[j] = ParallelDescriptor::Asend
1056  (send_data[j], send_size[j],
1057  ParallelContext::global_to_local_rank(send_rank[j]),
1058  SeqNum,
1059  ParallelContext::CommunicatorSub()).req();
1060  }
1061  else
1062  {
1063  ParallelDescriptor::Send
1064  (send_data[j], send_size[j],
1065  ParallelContext::global_to_local_rank(send_rank[j]),
1066  SeqNum,
1067  ParallelContext::CommunicatorSub());
1068  }
1069  }
1070 
1071  ++send_counter;
1072  }
1073  }
1074  }
1075 
1076 #ifdef BL_USE_MPI3
1077  if (ParallelDescriptor::MPIOneSided()) {
1078  if (N_rcvs > 0) MPI_Group_free(&rgroup);
1079  if (N_snds > 0) MPI_Group_free(&sgroup);
1080  }
1081 #endif
1082 
1083  //
1084  // Do the local work. Hope for a bit of communication/computation overlap.
1085  //
1086  if (ParallelDescriptor::TeamSize() > 1 && thecpc.m_threadsafe_loc)
1087  {
1088 #ifdef BL_USE_TEAM
1089 #ifdef _OPENMP
1090 #pragma omp parallel if (FAB::isCopyOMPSafe() && Gpu::notInLaunchRegion())
1091 #endif
1092  ParallelDescriptor::team_for(0, N_locs, [&] (int j)
1093  {
1094  const CopyComTag& tag = (*thecpc.m_LocTags)[j];
1095  // avoid self copy or plus
1096  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {
1097  auto const sfab = src.array(tag.srcIndex);
1098  auto dfab = this->array(tag.dstIndex);
1099  Dim3 offset = (tag.dbox.smallEnd() - tag.sbox.smallEnd()).dim3();
1100 
1101  if (op == FabArrayBase::COPY) {
1102  AMREX_HOST_DEVICE_FOR_4D ( tag.sbox, NC, i, j, k, n,
1103  {
1104  dfab(i+offset.x,j+offset.y,k+offset.z,DC+n) = sfab(i,j,k,SC+n);
1105  });
1106  } else {
1107  AMREX_HOST_DEVICE_FOR_4D ( tag.sbox, NC, i, j, k, n,
1108  {
1109  dfab(i+offset.x,j+offset.y,k+offset.z,DC+n) += sfab(i,j,k,SC+n);
1110  });
1111  }
1112  }
1113  });
1114 #ifdef AMREX_USE_GPU
1115  Gpu::Device::synchronize();
1116 #endif
1117 #endif
1118  }
1119  else
1120  {
1121  bool is_thread_safe = FAB::isCopyOMPSafe() && thecpc.m_threadsafe_loc;
1122  if (Gpu::inLaunchRegion() || !is_thread_safe)
1123  {
1124  // gpu version or omp over dest fabs
1125 
1126  LayoutData<Vector<FabCopyTag<FAB> > > copy_tags(boxArray(),DistributionMap());
1127  for (int j = 0; j < N_locs; ++j) {
1128  const CopyComTag& tag = (*thecpc.m_LocTags)[j];
1129  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {
1130  copy_tags[tag.dstIndex].push_back
1131  ({src.fabHostPtr(tag.srcIndex), tag.dbox, tag.sbox.smallEnd()-tag.dbox.smallEnd()});
1132  }
1133  }
1134 
1135 #ifdef _OPENMP
1136 #pragma omp parallel if (Gpu::notInLaunchRegion())
1137 #endif
1138  for (MFIter mfi(*this); mfi.isValid(); ++mfi) {
1139  auto dfab = this->array(mfi);
1140  const auto& fab_copy_tags = copy_tags[mfi];
1141  if (op == FabArrayBase::COPY) {
1142  for (auto const& fab_tag : fab_copy_tags) {
1143  Dim3 offset = fab_tag.offset.dim3();
1144  auto const sfab = fab_tag.sfab->array();
1145  AMREX_HOST_DEVICE_FOR_4D ( fab_tag.dbox, NC, i, j, k, n,
1146  {
1147  dfab(i,j,k,DC+n) = sfab(i+offset.x,j+offset.y,k+offset.z,SC+n);
1148  });
1149  }
1150  } else {
1151  for (auto const& fab_tag : fab_copy_tags) {
1152  Dim3 offset = fab_tag.offset.dim3();
1153  auto const sfab = fab_tag.sfab->array();
1154  AMREX_HOST_DEVICE_FOR_4D ( fab_tag.dbox, NC, i, j, k, n,
1155  {
1156  dfab(i,j,k,DC+n) += sfab(i+offset.x,j+offset.y,k+offset.z,SC+n);
1157  });
1158  }
1159  }
1160  }
1161  }
1162  else
1163  {
1164  // cpu version, omp over dest tile boxes
1165 #ifdef _OPENMP
1166 #pragma omp parallel for
1167 #endif
1168  for (int j = 0; j < N_locs; ++j)
1169  {
1170  const CopyComTag& tag = (*thecpc.m_LocTags)[j];
1171  // avoid self copy or plus
1172  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {
1173  const FAB* sfab = &(src[tag.srcIndex]);
1174  FAB* dfab = &(get(tag.dstIndex));
1175  if (op == FabArrayBase::COPY) {
1176  dfab->copy(*sfab, tag.sbox, SC, tag.dbox, DC, NC);
1177  } else {
1178  dfab->plus(*sfab, tag.sbox, tag.dbox, SC, DC, NC);
1179  }
1180  }
1181  }
1182  }
1183  }
1184 
1185  // prepare for receive
1186  Vector<const CopyComTagsContainer*> recv_cctc;
1187  LayoutData<Vector<VoidCopyTag> > recv_copy_tags;
1188  if (N_rcvs > 0)
1189  {
1190  recv_cctc.resize(N_rcvs,nullptr);
1191  for (int k = 0; k < N_rcvs; ++k)
1192  {
1193  if (recv_size[k] > 0)
1194  {
1195  auto const& cctc = thecpc.m_RcvTags->at(recv_from[k]);
1196  recv_cctc[k] = &cctc;
1197  }
1198  }
1199 
1200  bool is_thread_safe = FAB::isCopyOMPSafe() && thecpc.m_threadsafe_rcv;
1201  if (Gpu::inLaunchRegion() || !is_thread_safe)
1202  {
1203  recv_copy_tags.define(boxArray(),DistributionMap());
1204  for (int k = 0; k < N_rcvs; ++k) {
1205  const char* dptr = recv_data[k];
1206  if (dptr != nullptr)
1207  {
1208  auto const& cctc = *recv_cctc[k];
1209  for (auto const& tag : cctc)
1210  {
1211  recv_copy_tags[tag.dstIndex].push_back({dptr,tag.dbox});
1212  dptr += tag.dbox.numPts() * NC * sizeof(value_type);
1213  }
1214  BL_ASSERT(dptr == recv_data[k] + recv_size[k]);
1215  }
1216  }
1217  }
1218  }
1219 
1220  //
1221  // wait and unpack
1222  //
1223  if (ParallelDescriptor::MPIOneSided()) {
1224 #if defined(BL_USE_MPI3)
1225  if (N_snds > 0) MPI_Win_complete(ParallelDescriptor::cp_win);
1226  if (N_rcvs > 0) MPI_Win_wait (ParallelDescriptor::cp_win);
1227 #endif
1228  } else {
1229  if (actual_n_rcvs > 0) {
1230  Vector<MPI_Status> stats(N_rcvs);
1231  ParallelDescriptor::Waitall(recv_reqs, stats);
